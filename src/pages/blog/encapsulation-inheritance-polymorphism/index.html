Vagrant.configure("2") do |config|

  config.vm.box = "debian/jessie64"
  config.vm.network :private_network, ip: "192.168.100.251"

  config.vm.hostname = 'apa.dev'

  config.vm.provider :virtualbox do |vb|
    vb.name = "apa.dev-v2"
    vb.memory = 2048
  end

  if Vagrant.has_plugin?("vagrant-gatling-rsync")
    config.gatling.rsync_on_startup = false
  end

  config.ssh.forward_agent = true

  config.vm.provision "shell",
    inline: '[ -e /usr/local/bin/safesync ] && : || ln -sF /vagrant/safesync /usr/local/bin/safesync', run: 'once'

  config.vm.provision "shell",
    inline: '[ -e /usr/local/bin/safesync ] && chmod og+x /usr/local/bin/safesync', run: 'once'

  config.vm.provision :docker
  config.vm.provision :docker_compose, yml: "/vagrant/docker-compose.yml", run: "always", compose_version: "1.7.1"

  config.vm.provision :shell,
    inline: "[ ! -d /mnt/files ] && mkdir -p /mnt/files"

  config.vm.provision :shell,
    inline: "[ -e /mnt/files ] && chmod -R og+rwx /mnt/files", run: "always"

  config.vm.synced_folder ".", "/vagrant", type: "nfs"
  config.vm.synced_folder "./drupal", "/mnt/code/drupal", type: "rsync",
    rsync__exclude: %w(.git sites/default/files)
end
